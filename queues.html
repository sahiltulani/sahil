 1) PROGRAM :- IMPLEMENT THE OPERATIONS OF SIMPLE QUEUE IN AN INTEGER ARRAY(INSERT, DELETE, PEEK, ISEMPTY,ISFULL) 

     
    #include<stdio.h> 
    #include<stdbool.h> 

    int queue[10];
int front = -1;
int rear = -1;

// Function to check if the queue is empty
bool isEmpty()
{
    return front == -1;
}

// Function to check if the queue is full
bool isFull()
{
    return (rear == 10 - 1) ? true : false;
}

// Function to insert (enqueue) an element into the queue
void insert(int value)
{
    if (isFull())
    {
        printf("Queue is full. Cannot insert %d.\n", value);
        return;
    }

    if (isEmpty())
    {
        front = rear = 0;
    }
    else
    {
        rear++;
    }

    queue[rear] = value;
    printf("Inserted %d into the queue.\n", value);
}

// Function to delete (dequeue) an element from the queue
int delete()
{
    if (isEmpty())
    {
        printf("Queue is empty. Cannot delete.\n");
        return -1;
    }

    int deletedValue = queue[front];
    if (front == rear)
    {
        front = rear = -1;
    }
    else
    {
        front++;
    }

    return deletedValue;
}

// Function to get the front element of the queue (PEEK)
int peek()
{
    if (isEmpty())
    {
        printf("Queue is empty. Cannot peek.\n");
        return -1;
    }

    return queue[front];
}

int main()
{
    insert(10);
    insert(20);
    insert(30);
    insert(40);

    printf("Front element: %d\n", peek());

    printf("Deleted element: %d\n", delete ());
    printf("Deleted element: %d\n", delete ());

    printf("Front element: %d\n", peek());

    insert(50);
    insert(60);

    while (!isEmpty())
    {
        printf("Deleted element: %d\n", delete ());
    }

    return 0;
}

 3) PROGRAM :- IMPLEMENT THE OPERATIONS OF QUEUE DYNAMICALLY OF INTEGERS(INSERT,DELETE,PEEK,ISEMPTY,ISFULL) 

         
        #include<stdio.h> 
        #include<stdbool.h>
        #include<stdlib.h>

        struct Node
{
    int data;
    struct Node *next;
};

struct Queue
{
    struct Node *front;
    struct Node *rear;
};

// Function to initialize an empty queue
struct Queue *createQueue()
{
    struct Queue *queue = (struct Queue *)malloc(sizeof(struct Queue));
    if (queue == NULL)
    {
        perror("Memory allocation error");
        exit(1);
    }
    queue->front = queue->rear = NULL;
    return queue;
}

// Function to check if the queue is empty
bool isEmpty(struct Queue *queue)
{
    return (queue->front == NULL);
}

// Function to insert (enqueue) an element into the queue
void insert(struct Queue *queue, int value)
{
    struct Node *newNode = (struct Node *)malloc(sizeof(struct Node));
    if (newNode == NULL)
    {
        perror("Memory allocation error");
        exit(1);
    }
    newNode->data = value;
    newNode->next = NULL;

    if (isEmpty(queue))
    {
        queue->front = queue->rear = newNode;
    }
    else
    {
        queue->rear->next = newNode;
        queue->rear = newNode;
    }

    printf("Inserted %d into the queue.\n", value);
}

// Function to delete (dequeue) an element from the queue
int delete(struct Queue *queue)
{
    if (isEmpty(queue))
    {
        printf("Queue is empty. Cannot delete.\n");
        return -1;
    }

    struct Node *temp = queue->front;
    int deletedValue = temp->data;
    queue->front = temp->next;
    free(temp);

    return deletedValue;
}

// Function to get the front element of the queue (PEEK)
int peek(struct Queue *queue)
{
    if (isEmpty(queue))
    {
        printf("Queue is empty. Cannot peek.\n");
        return -1;
    }

    return queue->front->data;
}

int main()
{
    struct Queue *queue = createQueue();

    insert(queue, 10);
    insert(queue, 20);
    insert(queue, 30);
    insert(queue, 40);

    printf("Front element: %d\n", peek(queue));

    printf("Deleted element: %d\n", delete (queue));
    printf("Deleted element: %d\n", delete (queue));

    printf("Front element: %d\n", peek(queue));

    insert(queue, 50);
    insert(queue, 60);

    while (!isEmpty(queue))
    {
        printf("Deleted element: %d\n", delete (queue));
    }

    // Free memory used by the queue
    free(queue);

    return 0;
}

                Inserted 40 into the queue.        
                Front element: 10

                Deleted element: 20
                Front element: 30
                Inserted 50 into the queue.        
                Inserted 60 into the queue.        
                Deleted element: 30
                Deleted element: 40
                Deleted element: 50

 5) PROGRAM :- IMPLEMENT THE OPERATIONS OF CIRCULAR QUEUE IN AN INTEGER ARRAY(INSERT,DELETE,PEEK,ISEMPTY,ISFULL) 

         
        #include<stdio.h>
        
        #define MAX_SIZE 10

int queue[MAX_SIZE];
int front = -1, rear = -1;

// Function to check if the queue is empty
int isEmpty()
{
    return (front == -1 && rear == -1);
}

// Function to check if the queue is full
int isFull()
{
    return ((rear + 1) % MAX_SIZE == front);
}

// Function to insert an element into the queue
void enqueue(int value)
{
    if (isFull())
    {
        printf("Queue is full. Cannot enqueue %d.\n", value);
        return;
    }

    if (isEmpty())
    {
        front = rear = 0;
    }
    else
    {
        rear = (rear + 1) % MAX_SIZE;
    }

    queue[rear] = value;
    printf("%d has been enqueued.\n", value);
}

// Function to remove an element from the queue
void dequeue()
{
    if (isEmpty())
    {
        printf("Queue is empty. Cannot dequeue.\n");
        return;
    }

    printf("%d has been dequeued.\n", queue[front]);

    if (front == rear)
    {
        front = rear = -1;
    }
    else
    {
        front = (front + 1) % MAX_SIZE;
    }
}

// Function to get the front element of the queue without removing it
int peek()
{
    if (isEmpty())
    {
        printf("Queue is empty. No element to peek.\n");
        return -1;
    }

    return queue[front];
}

int main()
{
    enqueue(1);
    enqueue(2);
    enqueue(3);

    printf("Front element: %d\n", peek());

    dequeue();
    enqueue(4);

    printf("Front element: %d\n", peek());

    dequeue();
    dequeue();
    dequeue();

    if (isEmpty())
    {
        printf("Queue is empty.\n");
    }

    return 0;
}

        
      
      
                
Output:- 

                  
                1 has been enqueued.
                2 has been enqueued.
                3 has been enqueued.
                Front element: 1
                1 has been dequeued.
                4 has been enqueued.
                Front element: 2
                2 has been dequeued.
                3 has been dequeued.
                4 has been dequeued.
                Queue is empty.

 7) PROGRAM :- IMPLEMENT THE OPERATIONS OF PRIORITY QUEUE IN AN INTEGER ARRAY(INSERT,DELETE,PEEK,ISEMPTY,ISFULL) 

         
        #include<stdio.h>
        
        #define MAX_SIZE 10

int priorityQueue[MAX_SIZE];
int rear = -1;

// Function to check if the priority queue is empty
int isEmpty()
{
    return (rear == -1);
}

// Function to check if the priority queue is full
int isFull()
{
    return (rear == MAX_SIZE - 1);
}

// Function to insert an element with its priority into the priority queue
void insert(int value)
{
    if (isFull())
    {
        printf("Priority queue is full. Cannot insert %d.\n", value);
        return;
    }

    int i;
    for (i = rear; i >= 0; i--)
    {
        if (value > priorityQueue[i])
        {
            priorityQueue[i + 1] = priorityQueue[i];
        }
        else
        {
            break;
        }
    }

    priorityQueue[i + 1] = value;
    rear++;
    printf("%d has been inserted.\n", value);
}

// Function to remove the highest-priority element from the priority queue
void delete()
{
    if (isEmpty())
    {
        printf("Priority queue is empty. Cannot delete.\n");
        return;
    }

    printf("%d has been deleted.\n", priorityQueue[rear]);
    rear--;
}

// Function to get the highest-priority element without removing it
int peek()
{
    if (isEmpty())
    {
        printf("Priority queue is empty. No element to peek.\n");
        return -1;
    }

    return priorityQueue[rear];
}

int main()
{
    insert(5);
    insert(10);
    insert(3);

    printf("Highest-priority element: %d\n", peek());

    delete ();
    insert(7);

    printf("Highest-priority element: %d\n", peek());

    delete ();
    delete ();
    delete ();

    if (isEmpty())
    {
        printf("Priority queue is empty.\n");
    }

    return 0;
}

}

{
    return (rear == MAX_SIZE - 1);

{
    if (isFull())
    {
        printf("Deque is full. Cannot insert at the front.\n");
        return;
    }

    if (isEmpty())
    {
        front = rear = 0;
    }
    else
    {
        front--;
        if (front < 0)
        {
            front = MAX_SIZE - 1;
        }
    }

    deque[front] = value;
    printf("%d has been inserted at the front.\n", value);
}

    {
        printf("Deque is full. Cannot insert at the rear.\n");

        front = rear = 0;

    else
    {
        rear++;

        #include<stdio.h>
        #include<string.h>

        #define MAX_SIZE 10
#define MAX_STRING_LENGTH 50

char deque[MAX_SIZE][MAX_STRING_LENGTH];
int front = -1, rear = -1;

// Function to check if the deque is empty
int isEmpty()
{
    return (front == -1 && rear == -1);
}

// Function to check if the deque is full
int isFull()
{
    return (rear == MAX_SIZE - 1);
}

// Function to insert a string at the front of the deque
void insertFront(const char *value)
{
    if (isFull())
    {
        printf("Deque is full. Cannot insert at the front.\n");
        return;
    }

    if (isEmpty())
    {
        front = rear = 0;
    }
    else
    {
        front--;
        if (front < 0)
        {
            front = MAX_SIZE - 1;
        }
    }

    strcpy(deque[front], value);
    printf("%s has been inserted at the front.\n", value);
}

// Function to insert a string at the rear of the deque
void insertRear(const char *value)
{
    if (isFull())
    {
        printf("Deque is full. Cannot insert at the rear.\n");
        return;
    }

    if (isEmpty())
    {
        front = rear = 0;
    }
    else
    {
        rear++;
    }

    strcpy(deque[rear], value);
    printf("%s has been inserted at the rear.\n", value);
}

// Function to delete a string from the front of the deque
void deleteFront()
{
    if (isEmpty())
    {
        printf("Deque is empty. Cannot delete from the front.\n");
        return;
    }

    printf("%s has been deleted from the front.\n", deque[front]);

    if (front == rear)
    {
        front = rear = -1;
    }
    else
    {
        front++;
        if (front == MAX_SIZE)
        {
            front = 0;
        }
    }
}

// Function to delete a string from the rear of the deque
void deleteRear()
{
    if (isEmpty())
    {
        printf("Deque is empty. Cannot delete from the rear.\n");
        return;
    }

    printf("%s has been deleted from the rear.\n", deque[rear]);

    if (front == rear)
    {
        front = rear = -1;
    }
    else
    {
        rear--;
        if (rear < 0)
        {
            rear = MAX_SIZE - 1;
        }
    }
}

// Function to peek at the string at the front of the deque
const char *peekFront()
{
    if (isEmpty())
    {
        printf("Deque is empty. No element to peek at the front.\n");
        return NULL;
    }

    return deque[front];
}

// Function to peek at the string at the rear of the deque
const char *peekRear()
{
    if (isEmpty())
    {
        printf("Deque is empty. No element to peek at the rear.\n");
        return NULL;
    }

    return deque[rear];
}

int main()
{
    insertFront("Apple");
    insertRear("Banana");
    insertFront("Cherry");

    printf("Front element: %s, Rear element: %s\n", peekFront(), peekRear());

    deleteFront();
    deleteRear();

    printf("Front element: %s, Rear element: %s\n", peekFront(), peekRear());

    deleteFront();

    if (isEmpty())
    {
        printf("Deque is empty.\n");
    }

    return 0;
}


        
      
      
                
Output:-

                Banana has been inserted at the rear.      
                Cherry has been inserted at the front.     
                Front element: Cherry, Rear element: Banana
                Cherry has been deleted from the front.    
                Banana has been deleted from the rear.     
                Front element: Apple, Rear element: Apple  
                Apple has been deleted from the front.     
                Deque is empty.
