      #include<stdio.h>

      int isPalindrome(char *str)
{
    int left = 0;
    int right = 0;

    
    while (str[right] != '\0')
    {
        right++;
    }
    right--;

    
    while (left < right)
    {
        if (str[left] != str[right])
        {
            return 0; 
        }
        left++;
        right--;
    }
    return 1; 
}

int main()
{

    printf("Enter a string: ");
    scanf("%s", str);

    if (isPalindrome(str))
    {
        printf("%s is a palindrome.\n", str);
    }
    else
    {
        printf("%s is not a palindrome.\n", str);
    }

    return 0;
}

      #include<stdio.h>

      int main()
{
    char inputString[100];
    float floatValue = 0.0f;
    float decimalPlace = 0.1f;
    int isNegative = 0;   // To handle negative numbers
    int isFractional = 0; // To handle fractional parts

    // Prompt the user to enter a string
    printf("Enter a string: ");
    fgets(inputString, sizeof(inputString), stdin);

    int i = 0;

    // Check for a leading minus sign to handle negative numbers
    if (inputString[i] == '-')
    {
        isNegative = 1;
        i++;
    }

    // Iterate through the string and convert it to a float
    while (inputString[i] != '\0')
    {
        char c = inputString[i];

        if (c >= '0' && c <= '9')
        {
            if (isFractional)
            {
                floatValue = floatValue + (c - '0') * decimalPlace;
                decimalPlace /= 10;
            }
            else
            {
                floatValue = floatValue * 10 + (c - '0');
            }
        }
        else if (c == '.')
        {
            isFractional = 1;
        }
        else
        {
            break; // Exit the loop if a non-numeric character is encountered
        }
        i++;
    }

    if (isNegative)
    {
        floatValue = -floatValue;
    }

    printf("Float value: %f\n", floatValue);

    return 0;
}

 21) PROGRAM :- PRINT ALL THE PERMUTATION OF A STRING I.E. "ABCDE". 

        #include<stdio.h>

        // Function to swap characters at positions i and j in the string
void swap(char *str, int i, int j)
{
    char temp = str[i];
    str[i] = str[j];
    str[j] = temp;
}

// Function to generate all permutations of a string
void permute(char *str, int l, int r)
{
    if (l == r)
    {
        printf("%s\n", str); // Print the permutation when the string is fully permuted
    }
    else
    {
        for (int i = l; i <= r; i++)
        {
            swap(str, l, i);        // Swap characters at positions l and i
            permute(str, l + 1, r); // Recursively permute the remaining characters
            swap(str, l, i);        // Restore the original string for the next iteration
        }
    }
}

int main()
{
    char str[] = "ABCDE";
    int n = strlen(str);

    printf("All permutations of the string \"%s\":\n", str);
    permute(str, 0, n - 1);

    return 0;
}

 22) PROGRAM :- ENTER THE STRING FROM THE USER, SORT THE STRING IN ASCENDING ORDER AND PRINT IT. 

        
        #include<stdio.h>
        #include<string.h>

        void bubbleSort(char str[], int n)
{
    int i, j;
    for (i = 0; i < n - 1; i++)
    {
        for (j = 0; j < n - i - 1; j++)
        {
            if (str[j] > str[j + 1])
            {
                // Swap the characters if they are out of order
                char temp = str[j];
                str[j] = str[j + 1];
                str[j + 1] = temp;
            }
        }
    }
}

int main()
{
    char str[100]; // You can adjust the size as needed
    printf("Enter a string: ");
    scanf("%s", str);

    int n = strlen(str);

    bubbleSort(str, n);

    printf("Sorted string in ascending order: %s\n", str);

    return 0;
}

        int countWordOccurrences(const char *sentence, const char *word)
{
    int count = 0;
    const char *pos = sentence;

    while ((pos = strstr(pos, word)) != NULL)
    {
        count++;
        pos += strlen(word);
    }

    return count;
}

{

    char word[100];      // You can adjust the size as needed

        void replaceWord(char *sentence, const char *oldWord, const char *newWord)
{
    char *pos = sentence;
    int oldWordLength = strlen(oldWord);
    int newWordLength = strlen(newWord);
    int diff = newWordLength - oldWordLength;

    while ((pos = strstr(pos, oldWord)) != NULL)
    {
        memmove(pos + newWordLength, pos + oldWordLength, strlen(pos + oldWordLength) + 1);
        memcpy(pos, newWord, newWordLength);
        pos += newWordLength;
        sentence = realloc(sentence, strlen(sentence) + diff + 1);
    }
}

int main()
{
    char sentence[1000]; // You can adjust the size as needed
    char oldWord[100];   // You can adjust the size as needed
    char newWord[100];   // You can adjust the size as needed

    printf("Enter a sentence: ");
    fgets(sentence, sizeof(sentence), stdin);
    sentence[strlen(sentence) - 1] = '\0'; // Remove the newline character at the end

    printf("Enter the word to replace: ");
    scanf("%s", oldWord);

    printf("Enter the new word: ");
    scanf("%s", newWord);

    replaceWord(sentence, oldWord, newWord);

    printf("Modified sentence: %s\n", sentence);

    return 0;
}

        #include<stdio.h>
        #include<string.h>
        #include<stdbool.h>

        void trimSpaces(char *str)
{
    int len = strlen(str);
    int start = 0, end = len - 1;

    // Find the first non-space character from the beginning
    while (start < len && isspace(str[start]))
    {
        start++;
    }

    // Find the first non-space character from the end
    while (end >= 0 && isspace(str[end]))
    {
        end--;
    }

    // If the string is all spaces, make it an empty string
    if (start > end)
    {
        str[0] = '\0';
    }
    else
    {
        // Shift the characters to the beginning of the string
        int i, j = 0;
        for (i = start; i <= end; i++, j++)
        {
            str[j] = str[i];
        }
        str[j] = '\0'; // Null-terminate the trimmed string
    }
}

int main()
{
    char str[1000]; // You can adjust the size as needed

    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin);
    str[strlen(str) - 1] = '\0'; // Remove the newline character at the end

    trimSpaces(str);

    printf("Trimmed string: %s\n", str);

    return 0;
}
