 1) PROGRAM :- SUM OF MATRIX 

   
  #include<stdio.h> 
  #include<conio.h>
  #include<dos.h>
                
void main()
{
  int a[2][3], b[2][3], sum[2][3], i, j;

  printf("Enter first Matrix:- \n");
  for (i = 0; i < 2; i++)
  {
    for (j = 0; j < 3; j++)
    {
      scanf("%d", &a[i][j]);
    }
  }

  printf("Enter Second Matrix:- \n");
  for (i = 0; i < 2; i++)
  {
    for (j = 0; j < 3; j++)
    {
      scanf("%d", &b[i][j]);
    }
  }

  printf("The first Matrix:- \n");
  for (i = 0; i < 2; i++)
  {
    for (j = 0; j < 3; j++)
    {
      printf("%d \t", a[i][j]);
    }
    printf("\n");
  }

  printf("The Second Matrix:- \n");
  for (i = 0; i < 2; i++)
  {
    for (j = 0; j < 3; j++)
    {
      printf("%d \t", b[i][j]);
    }
    printf("\n");
  }

  printf("Sum of Matrix:- \n");
  for (i = 0; i < 2; i++)
  {
    for (j = 0; j < 3; j++)
    {
      sum[i][j] = a[i][j] + b[i][j];
      printf("%d \t", sum[i][j]);
    }
    printf("\n");
  }
  getch();
}

 2) Program:- Multiplication of Matrix

   
  #include<stdio.h> 
  #include<conio.h>
  #include<dos.h>

    
// Multiplication of Matrix

void main()
{
  int a[3][3], b[3][2], result[3][2];
  int sum = 0;
  int i, j, k;

  clrscr();

  printf("enter first matrix:- \n");
  for (i = 0; i < 3; i++)
  {
    for (j = 0; j < 3; j++)
    {
      scanf("%d", &a[i][j]);
    }
  }

  printf("enter second matrix:- \n");
  for (i = 0; i < 3; i++)
  {
    for (j = 0; j < 2; j++)
    {
      scanf("%d", &b[i][j]);
    }
  }

  printf("The first matrix:- \n");
  for (i = 0; i < 3; i++)
  {
    for (j = 0; j < 3; j++)
    {
      printf("%d \t", a[i][j]);
    }
    printf("\n");
  }

  printf("The Second matrix:- \n");
  for (i = 0; i < 3; i++)
  {
    for (j = 0; j < 2; j++)
    {
      printf("%d \t", b[i][j]);
    }
    printf("\n");
  }

  // calculate matrix

  for (i = 0; i < 3; i++)
  {
    for (j = 0; j < 2; j++)
    {
      for (k = 0; k < 3; k++)
      {
        sum = sum + a[i][k] * b[k][j];
      }
      result[i][j] = sum;
    }
  }

  printf("Multiplication of Matrix:- \n");

  for (i = 0; i < 3; i++)
  {
    for (j = 0; j < 2; j++)
    {
      printf("%d \t", result[i][j]);
    }
    printf("\n");
  }

  delay(8000);
}

  {
    for (j = 0; j < 3; j++)
    {
      printf("%d \t", a[i][j]);
    }
    printf("\n");
  }

  printf("The Transpose matrix is :- \n");
  for (i = 0; i < 3; i++)
  {
    for (j = 0; j < 3; j++)

 4) Program:- Subtraction of Matrix

     
  #include<stdio.h> 
  #include<conio.h>
  #include<dos.h>

  // Subtraction of Matrix

void main()
{
  int a[2][3], b[2][3], sub[2][3], i, j;
  clrscr();

  printf("Enter first Matrix:- \n");
  for (i = 0; i < 2; i++)
  {
    for (j = 0; j < 3; j++)
    {
      scanf("%d", &a[i][j]);
    }
  }

  printf("Enter Second Matrix:- \n");
  for (i = 0; i < 2; i++)
  {
    for (j = 0; j < 3; j++)
    {
      scanf("%d", &b[i][j]);
    }
  }

  printf("The first Matrix:- \n");
  for (i = 0; i < 2; i++)
  {
    for (j = 0; j < 3; j++)
    {
      printf("%d \t", a[i][j]);
    }
    printf("\n");
  }

  printf("The Second Matrix:- \n");
  for (i = 0; i < 2; i++)
  {
    for (j = 0; j < 3; j++)
    {
      printf("%d \t", b[i][j]);
    }
    printf("\n");
  }

  printf("Subtraction of Matrix:- \n");
  for (i = 0; i < 2; i++)
  {
    for (j = 0; j < 3; j++)
    {
      sub[i][j] = a[i][j] - b[i][j];
      printf("%d \t", sub[i][j]);
    }
    printf("\n");
  }

  getch();
}
