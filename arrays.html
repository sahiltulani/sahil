 9) PROGRAM :- Enter an array from user and find the k-th smallest value  

     
    #include<stdio.h> 

    int main()
{
    int n, i, k;
    int temp;

    printf("Enter the number of elements: ");
    scanf("%d", &n);

    int arr[n];

    printf("Enter the elements of the array:\n");
    for (i = 0; i < n; i++)
    {
        printf("Enter element %d: ", i + 1);
        scanf("%d", &arr[i]);
    }

    printf("Enter the value of k: ");
    scanf("%d", &k);

    // Using bubble sort to sort the array
    for (i = 0; i < n - 1; i++)
    {
        for (int j = 0; j < n - i - 1; j++)
        {
            if (arr[j] > arr[j + 1])
            {
                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }

    printf("The %dth smallest value is: %d\n", k, arr[k - 1]);

    return 0;
}

        void swap(int *a, int *b)
{
    int temp = *a;
    *a = *b;
    *b = temp;
}

void comb(int arr[], int n, int index)
{
    int i;

    if (index == n)
    {
        for (i = 0; i < n; i++)
        {
            printf("%d", arr[i]);
        }
        printf("\n");
        return;
    }

    for (i = index; i < n; i++)
    {
        swap(&arr[index], &arr[i]);
        comb(arr, n, index + 1);
        swap(&arr[index], &arr[i]);
    }
}

    #include<stdio.h>

    int isPalindrome(int num)
{
    int reversed = 0;
    int original = num;

    while (num > 0)
    {
        int remainder = num % 10;
        reversed = reversed * 10 + remainder;
        num /= 10;
    }

    return original == reversed;
}

int main()
{
    int n, i;

    printf("Enter the number of elements: ");

        scanf("%d", &arr[i]);
    }

    printf("Palindrome numbers in the array are: ");
    for (i = 0; i < n; i++)
    {
        if (isPalindrome(arr[i]))

 17) PROGRAM :- Merge the two sorted arrays into one array but in sorted order 

     
    #include<stdio.h>

    int main()
{
    int n, m, i, j, k;

    printf("Enter the number of elements in the first array: ");
    scanf("%d", &n);

    int arr1[n];

    printf("Enter the elements of the first array in sorted order:\n");
    for (i = 0; i < n; i++)
    {
        printf("Enter element %d: ", i + 1);
        scanf("%d", &arr1[i]);
    }

    printf("Enter the number of elements in the second array: ");
    scanf("%d", &m);

    int arr2[m];

    printf("Enter the elements of the second array in sorted order:\n");
    for (i = 0; i < m; i++)
    {
        printf("Enter element %d: ", i + 1);
        scanf("%d", &arr2[i]);
    }

    int merged[n + m];
    i = 0, j = 0, k = 0;

    while (i < n && j < m)
    {
        if (arr1[i] < arr2[j])
        {
            merged[k] = arr1[i];
            i++;
        }
        else
        {
            merged[k] = arr2[j];
            j++;
        }
        k++;
    }

    while (i < n)
    {
        merged[k] = arr1[i];
        i++;
        k++;
    }

    while (j < m)
    {
        merged[k] = arr2[j];
        j++;
        k++;
    }

    printf("Merged array in sorted order: ");
    for (i = 0; i < n + m; i++)
    {
        printf("%d ", merged[i]);
    }
    printf("\n");

    return 0;
}
