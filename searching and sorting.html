 1) PROGRAM :- ENTER A INTEGER ARRAY FROM THE USER, ENTER AN INTEGER VALUE AND PERFORM LINEAR SEARCH ALGORITHM 
 THROUGH ITERATION.  

     
    #include<stdio.h>
    
    int main()
{
    int n, searchValue, found = 0;

    printf("Enter the number of elements in the array: ");
    scanf("%d", &n);

    int arr[n];

    printf("Enter the elements of the array:\n");
    for (int i = 0; i < n; i++)
    {
        scanf("%d", &arr[i]);
    }

    printf("Enter the integer value to search for: ");
    scanf("%d", &searchValue);

    for (int i = 0; i < n; i++)
    {
        if (arr[i] == searchValue)
        {
            found = 1;
            printf("Element %d found at index %d.\n", searchValue, i);
            break; // Exit the loop once the element is found.
        }
    }

    if (!found)
    {
        printf("Element %d not found in the array.\n", searchValue);
    }

    return 0;
}

 2) PROGRAM :- ENTER A STRING FROM THE USER, ENTER A CHARACTER AND PERFORM LINEAR SEARCH ALGORITHM THROUGH 
 RECURSION.  

       
      #include<stdio.h>

      int linearSearchRecursive(char str[], char target, int index)
{
    if (str[index] == '\0')
    {
        return -1;
    }

    if (str[index] == target)
    {
        return index;
    }

    return linearSearchRecursive(str, target, index + 1);
}

int main()
{
    char str[100], target;

    printf("Enter a string: ");
    scanf("%s", str);

    printf("Enter the character to search for: ");
    scanf(" %c", &target); 

    int result = linearSearchRecursive(str, target, 0);

    if (result == -1)
    {
        printf("Character '%c' not found in the string.\n", target);
    }
    else
    {
        printf("Character '%c' found at index %d in the string.\n", target, result);
    }

    return 0;
}

      
     
    
    
              
Output:- 

                
              Enter a string: hello
              Enter the character to search for: e
              Character 'e' found at index 1 in the string.

      #include<stdio.h>

      int linearSearchIterative(char str[], char target)
{
    int index = 0;
    while (str[index] != '\0')
    {
        if (str[index] == target)
        {
            return index;
        }
        index++;
    }
    return -1;
}

 5) PROGRAM :- ENTER A INTEGER ARRAY FROM THE USER, ENTER AN INTEGER VALUE AND PERFORM BINARY SEARCH ALGORITHM 
THROUGH ITERATION.  

       
      #include<stdio.h>

      int binarySearchIterative(int arr[], int n, int searchValue)
{
    int left = 0;
    int right = n - 1;

    while (left <= right)
    {
        int mid = left + (right - left) / 2;

        if (arr[mid] == searchValue)
        {
            return mid;
        }

        if (arr[mid] < searchValue)
        {
            left = mid + 1;
        }
        else
        {
            right = mid - 1;
        }
    }

    return -1;
}

int main()
{
    int n, searchValue;

    printf("Enter the number of elements in the sorted array: ");
    scanf("%d", &n);

    int arr[n];

    printf("Enter the elements of the sorted array:\n");
    for (int i = 0; i < n; i++)
    {
        scanf("%d", &arr[i]);
    }

    printf("Enter the integer value to search for: ");
    scanf("%d", &searchValue);

    int result = binarySearchIterative(arr, n, searchValue);

    if (result == -1)
    {
        printf("Element %d not found in the array.\n", searchValue);
    }
    else
    {
        printf("Element %d found at index %d in the array.\n", searchValue, result);
    }

    return 0;
}

 6) PROGRAM :- ENTER A STRING FROM THE USER, ENTER A CHARACTER AND PERFORM BINARY SEARCH ALGORITHM THROUGH 
 RECURSION.  

       
      #include<stdio.h>
      #include<string.h>


      int binarySearchRecursive(char str[], char target, int left, int right)
{
    if (left > right)
    {
        return -1;
    }

    int mid = left + (right - left) / 2;

    if (str[mid] == target)
    {
        return mid;
    }

    if (str[mid] < target)
    {
        return binarySearchRecursive(str, target, mid + 1, right);
    }
    else
    {
        return binarySearchRecursive(str, target, left, mid - 1);
    }
}

int main()
{
    char str[100], target;

    printf("Enter a sorted string: ");
    scanf("%s", str);

    printf("Enter the character to search for: ");
    scanf(" %c", &target);

    int result = binarySearchRecursive(str, target, 0, strlen(str) - 1);

    if (result == -1)
    {
        printf("Character '%c' not found in the string.\n", target);
    }
    else
    {
        printf("Character '%c' found at index %d in the string.\n", target, result);
    }

    return 0;
}
