      // Define the stack structure
struct CharStack

    char data[100];

    stack->top = -1;
}

// Function to check if the stack is empty

    return (stack->top == -1);

// Function to check if the stack is full

    return (stack->top == 100 - 1);

void push(struct CharStack *stack, char value)
{
    if (isFull(stack))
    {
        printf("Stack is full. Cannot push %c\n", value);
    }
    else
    {
        stack->data[++stack->top] = value;
        printf("Pushed %c onto the stack\n", value);
    }
}

// Function to pop a character from the stack
char pop(struct CharStack *stack)
{
    if (isEmpty(stack))
    {
        printf("Stack is empty. Cannot pop.\n");
        return '\0'; // Return a placeholder value to indicate an error
    }
    else
    {
        char value = stack->data[stack->top--];
        return value;
    }
}

// Function to peek at the top element of the stack without removing it
char peek(struct CharStack *stack)
{
    if (isEmpty(stack))
    {
        printf("Stack is empty. Cannot peek.\n");
        return '\0'; // Return a placeholder value to indicate an error
    }
    else
    {
        return stack->data[stack->top];
    }
}

{
    struct CharStack stack;
    initialize(&stack);

    push(&stack, 'A');
    push(&stack, 'B');
    push(&stack, 'C');

    while (!isEmpty(&stack))

        printf("Popped: %c\n", pop(&stack));

    return 0;

 3) PROGRAM :-  CREATE A STACK IMPLEMENTING THE OPERATIONS ON AN INTEGER NODES (DYNAMICALLY) (PUSH, POP, PEEK, 
 ISEMPTY, ISFULL) 

         
        #include<stdio.h>
        #include<stdlib.h>

        // Define the structure for a stack node
struct Node
{
    int data;
    struct Node *next;
};

// Define a stack structure
struct Stack
{
    struct Node *top;
    unsigned int size;
};

// Function to create an empty stack
struct Stack *createStack()
{
    struct Stack *stack = (struct Stack *)malloc(sizeof(struct Stack));
    stack->top = NULL;
    stack->size = 0;
    return stack;
}

// Function to check if the stack is empty
int isEmpty(struct Stack *stack)
{
    return (stack->top == NULL);
}

// Function to push an element onto the stack
void push(struct Stack *stack, int data)
{
    struct Node *newNode = (struct Node *)malloc(sizeof(struct Node));
    newNode->data = data;
    newNode->next = stack->top;
    stack->top = newNode;
    stack->size++;
}

// Function to pop an element from the stack
int pop(struct Stack *stack)
{
    if (isEmpty(stack))
    {
        printf("Stack is empty. Cannot pop.\n");
        return -1; // Return a sentinel value to indicate an error
    }
    struct Node *temp = stack->top;
    int poppedValue = temp->data;
    stack->top = temp->next;
    free(temp);
    stack->size--;
    return poppedValue;
}

// Function to get the top element without popping it
int peek(struct Stack *stack)
{
    if (isEmpty(stack))
    {
        printf("Stack is empty. Cannot peek.\n");
        return -1; // Return a sentinel value to indicate an error
    }
    return stack->top->data;
}

int main()
{
    struct Stack *stack = createStack();

    // Push elements onto the stack
    push(stack, 10);
    push(stack, 20);
    push(stack, 30);

    printf("Top element: %d\n", peek(stack));

    // Pop elements from the stack
    printf("Popped element: %d\n", pop(stack));
    printf("Popped element: %d\n", pop(stack));

    printf("Is the stack empty? %s\n", isEmpty(stack) ? "Yes" : "No");

    return 0;
}

  
        
      
      
                
Output:- 

                  
                Top element: 30
                Popped element: 30
                Popped element: 20
                Is the stack empty? No

 5) PROGRAM :-  ENTER THE STRING FROM THE USER AND REVERSE THE STRING USING STACKS

    char data;
    struct Node *next;

    struct Node *top;

struct Stack *createStack()

    struct Stack *stack = (struct Stack *)malloc(sizeof(struct Stack));
    stack->top = NULL;
    stack->size = 0;
    return stack;

// Function to check if the stack is empty

    return (stack->top == NULL);

void push(struct Stack *stack, char data)
{
    struct Node *newNode = (struct Node *)malloc(sizeof(struct Node));
    newNode->data = data;
    newNode->next = stack->top;
    stack->top = newNode;
    stack->size++;
}

// Function to pop a character from the stack
char pop(struct Stack *stack)

    if (isEmpty(stack))
    {
        return '\0'; // Return a sentinel value to indicate an error
    }
    struct Node *temp = stack->top;
    char poppedValue = temp->data;
    stack->top = temp->next;
    free(temp);
    stack->size--;
    return poppedValue;
}

{
    struct Stack *stack = createStack();

    printf("Enter a string: ");
    char inputString[100];
    fgets(inputString, sizeof(inputString), stdin);

    // Push each character of the input string onto the stack
    for (int i = 0; inputString[i] != '\0' && inputString[i] != '\n'; i++)
    {
        push(stack, inputString[i]);
    }

    // Pop characters from the stack to reverse the string
    char reversedString[100];
    int index = 0;
    while (!isEmpty(stack))
    {
        reversedString[index] = pop(stack);
        index++;
    }
    reversedString[index] = '\0';

    printf("Reversed string: %s\n", reversedString);

    return 0;

 6) PROGRAM :-  CREATE THE PROGRAM TO MATCH THE SINGLE BRACKET USING STACKS

        #include<stdlib.h>
        #include<stdbool.h>


        // Define the structure for a stack node
struct Node
{
    char data;
    struct Node *next;
};

// Define a stack structure
struct Stack
{
    struct Node *top;
    unsigned int size;

// Function to create an empty stack
struct Stack *createStack()

    return stack;

// Function to check if the stack is empty
int isEmpty(struct Stack *stack)

// Function to push a character onto the stack
void push(struct Stack *stack, char data)
{
    struct Node *newNode = (struct Node *)malloc(sizeof(struct Node));
    newNode->data = data;
    newNode->next = stack->top;
    stack->top = newNode;
    stack->size++;
}

// Function to pop a character from the stack
char pop(struct Stack *stack)
{
    if (isEmpty(stack))
    {
        return '\0'; // Return a sentinel value to indicate an error
    }
    struct Node *temp = stack->top;
    char poppedValue = temp->data;
    stack->top = temp->next;
    free(temp);
    stack->size--;
    return poppedValue;
}

// Function to check if two characters form a matching bracket pair
bool isMatchingPair(char opening, char closing)
{
    if (opening == '(' && closing == ')')
        return true;
    if (opening == '[' && closing == ']')
        return true;
    if (opening == '{' && closing == '}')
        return true;
    return false;
}

// Function to check if the input string has balanced brackets
bool isBalanced(const char *inputString)
{
    struct Stack *stack = createStack();

    for (int i = 0; inputString[i] != '\0'; i++)
    {
        if (inputString[i] == '(' || inputString[i] == '[' || inputString[i] == '{')
        {
            push(stack, inputString[i]);
        }
        else if (inputString[i] == ')' || inputString[i] == ']' || inputString[i] == '}')
        {
            if (isEmpty(stack) || !isMatchingPair(pop(stack), inputString[i]))
            {
                free(stack); // Free the stack before returning
                return false;
            }
        }
    }

    bool isBalanced = isEmpty(stack);
    free(stack); // Free the stack before returning
    return isBalanced;
}

int main()
{
    const char *inputString = "((a+b))";
    if (isBalanced(inputString))
    {
        printf("The input string has balanced brackets.\n");
    }
    else
    {
        printf("The input string does not have balanced brackets.\n");
    }

    return 0;
}

 7) PROGRAM :-  CREATE TWO INTEGER STACKS IN A SINGLE ARRAY 

         
        #include<stdio.h>
        #include<stdlib.h>
        

        struct DoubleStack
{
    int data[100];
    int top1;
    int top2;
};

// Initialize the double stack
void initDoubleStack(struct DoubleStack *dStack)
{
    dStack->top1 = -1;       // Initialize the top of the first stack
    dStack->top2 = 100; // Initialize the top of the second stack
}

// Check if the first stack is full
int isFullStack1(struct DoubleStack *dStack)
{
    return (dStack->top1 + 1 == dStack->top2);
}

// Check if the second stack is full
int isFullStack2(struct DoubleStack *dStack)
{
    return (dStack->top2 - 1 == dStack->top1);
}

// Push an element onto the first stack
void pushStack1(struct DoubleStack *dStack, int item)
{
    if (isFullStack1(dStack))
    {
        printf("Stack 1 is full. Cannot push.\n");
        return;
    }
    dStack->data[++dStack->top1] = item;
}

// Push an element onto the second stack
void pushStack2(struct DoubleStack *dStack, int item)
{
    if (isFullStack2(dStack))
    {
        printf("Stack 2 is full. Cannot push.\n");
        return;
    }
    dStack->data[--dStack->top2] = item;
}

// Pop an element from the first stack
int popStack1(struct DoubleStack *dStack)
{
    if (dStack->top1 == -1)
    {
        printf("Stack 1 is empty. Cannot pop.\n");
        return -1; // Return a sentinel value to indicate an error
    }
    return dStack->data[dStack->top1--];
}

// Pop an element from the second stack
int popStack2(struct DoubleStack *dStack)
{
    if (dStack->top2 == 100)
    {
        printf("Stack 2 is empty. Cannot pop.\n");
        return -1; // Return a sentinel value to indicate an error
    }
    return dStack->data[dStack->top2++];
}

int main()
{
    struct DoubleStack dStack;
    initDoubleStack(&dStack);

    pushStack1(&dStack, 1);
    pushStack1(&dStack, 2);
    pushStack2(&dStack, 11);
    pushStack2(&dStack, 12);

    printf("Popped from Stack 1: %d\n", popStack1(&dStack));
    printf("Popped from Stack 2: %d\n", popStack2(&dStack));

    return 0;
}

                Popped from Stack 1: 2
                Popped from Stack 2: 12
    
            
        
 8) PROGRAM :-  SORT THE STACK IN ASCENDING ORDER USING INTEGER ARRAY 

         
        #include<stdio.h>
        #include<stdlib.h>
        #include<stdbool.h>
        

        struct Stack
{
    int data[100];
    int top;
};

// Initialize the stack
void initStack(struct Stack *stack)
{
    stack->top = -1;
}

// Check if the stack is empty
bool isEmpty(struct Stack *stack)
{
    return (stack->top == -1);
}

// Push an element onto the stack
void push(struct Stack *stack, int item)
{
    if (stack->top == 100 - 1)
    {
        printf("Stack is full. Cannot push.\n");
        return;
    }
    stack->data[++stack->top] = item;
}

// Pop an element from the stack
int pop(struct Stack *stack)
{
    if (isEmpty(stack))
    {
        printf("Stack is empty. Cannot pop.\n");
        return -1; // Return a sentinel value to indicate an error
    }
    return stack->data[stack->top--];
}

// Sort the stack in ascending order
void sortStack(struct Stack *stack)
{
    struct Stack tempStack;
    initStack(&tempStack);

    while (!isEmpty(stack))
    {
        int item = pop(stack);
        while (!isEmpty(&tempStack) && item < tempStack.data[tempStack.top])
        {
            push(stack, pop(&tempStack));
        }
        push(&tempStack, item);
    }

    while (!isEmpty(&tempStack))
    {
        push(stack, pop(&tempStack));
    }
}

int main()
{
    struct Stack stack;
    initStack(&stack);

    push(&stack, 30);
    push(&stack, -5);
    push(&stack, 18);
    push(&stack, 14);
    push(&stack, -3);

    printf("Original Stack:\n");
    while (!isEmpty(&stack))
    {
        printf("%d\n", pop(&stack));
    }

    initStack(&stack); // Reset the stack

    push(&stack, 30);
    push(&stack, -5);
    push(&stack, 18);
    push(&stack, 14);
    push(&stack, -3);

    sortStack(&stack);

    printf("\nSorted Stack (in ascending order):\n");
    while (!isEmpty(&stack))
    {
        printf("%d\n", pop(&stack));
    }

    return 0;
}




      
      
                
Output:- 

                  
                Original Stack:
                -3
                14
                18
                -5
                30
                
                Sorted Stack (in ascending order):
                -5
                -3
                14
                18
                30
